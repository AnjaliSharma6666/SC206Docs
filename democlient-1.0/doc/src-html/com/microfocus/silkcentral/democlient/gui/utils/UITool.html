005package com.microfocus.silkcentral.democlient.gui.utils;
006
007import java.awt.BorderLayout;
008import java.awt.Component;
009import java.awt.Dimension;
010import java.awt.GridLayout;
011import java.awt.Toolkit;
012import java.awt.Window;
013
014import javax.swing.JButton;
015import javax.swing.JOptionPane;
016import javax.swing.JPanel;
017
018public class UITool {
019
020  private UITool() {
021
022  }
023
024  /**
025   * Sets the visibility of a (dialog) window to 'true' and centers the window.
026   * 
027   * @param window
028   *          The window to be shown and centered.
029   */
030  public static void centerAndShow(Window window) {
031    Toolkit toolkit = Toolkit.getDefaultToolkit();
032    Dimension screenSize = toolkit.getScreenSize();
033
034    int x = (screenSize.width - window.getWidth()) / 2;
035    int y = (screenSize.height - window.getHeight()) / 2;
036
037    window.setLocation(x, y);
038    window.setVisible(true);
039  }
040
041  /**
042   * Simple error handling. Shows the exception message within a dialog box.
043   * 
044   * @param parent
045   *          The frame in which the dialog is displayed. If null a default Frame is used.
046   * @param exception
047   *          Any exception thrown during handling error.
048   */
049  public static void handleError(Component parent, Exception exception) {
050    exception.printStackTrace();
051    JOptionPane.showMessageDialog(parent, exception.getMessage());
052  }
053
054  /**
055   * Returns a panel containing the specified buttons.
056   * 
057   * @param buttons
058   *          Buttons to be added to the created panel.
059   * @return a new panel component
060   */
061  public static Component createButtonsPane(JButton... buttons) {
062    JPanel panel = new JPanel(new BorderLayout());
063    JPanel grid = new JPanel(new GridLayout(1, -1, 5, 0));
064    for (JButton btn : buttons) {
065      grid.add(btn);
066    }
067    panel.add(grid, BorderLayout.EAST);
068    return panel;
069  }
070
071}
