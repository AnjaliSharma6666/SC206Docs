064
065  private boolean accepted;
066
067  public static enum Type {
068    CLOSE, ACCEPT
069  }
070
071  public AboutDialog(Frame parent, String javaDocURL, Type type) {
072    super(parent, true);
073    setTitle("About " + DemoClient.TITLE);
074    setDefaultCloseOperation(JDialog.DISPOSE_ON_CLOSE);
075
076    JPanel contentPn = (JPanel) getContentPane();
077    contentPn.setBackground(Color.WHITE);
078    contentPn.setLayout(new BorderLayout());
079    contentPn.setBorder(BorderFactory.createEmptyBorder(11, 11, 10, 10));
080
081    JButton closeBtn = new JButton();
082    closeBtn.addActionListener(new ActionListener() {
083      @Override
084      public void actionPerformed(ActionEvent e) {
085        accepted = true;
086        setVisible(false);
087        dispose();
088      }
089    });
090
091    JPanel buttonPn = new JPanel();
092    if (type == Type.ACCEPT) {
093      JButton declineBtn = new JButton("Decline");
094      declineBtn.addActionListener(new ActionListener() {
095        @Override
096        public void actionPerformed(ActionEvent e) {
097          System.exit(0);
098        }
099      });
100      buttonPn.add(declineBtn);
101      closeBtn.setText("Accept");
102    } else {
103      closeBtn.setText("Close");
104    }
105
106    buttonPn.add(closeBtn);
107    buttonPn.setBackground(Color.WHITE);
108
109    contentPn.add(createContent(javaDocURL), BorderLayout.CENTER);
110    contentPn.add(buttonPn, BorderLayout.SOUTH);
111
112    setSize(630, 470);
113  }
114
115  private Component createContent(String javaDocURL) {
116    String aboutMessage = ABOUT_MESSAGE.replaceAll("\\$\\{javaDocURL\\}", javaDocURL);
117    JEditorPane aboutEP = new JEditorPane("text/html; charset=UTF-8", aboutMessage);
118    aboutEP.setEditable(false);
119    aboutEP.setOpaque(false);
120
121    aboutEP.addHyperlinkListener(new HyperlinkListener() {
122      @Override
123      public void hyperlinkUpdate(HyperlinkEvent event) {
124        if (event.getEventType() == HyperlinkEvent.EventType.ACTIVATED) {
125          try {
126            Desktop.getDesktop().browse(event.getURL().toURI());
127          } catch (Exception e) {
128            UITool.handleError(AboutDialog.this, e);
129          }
130        }
131      }
132    });
133
134    return aboutEP;
135
136  }
137
138  public boolean accepted() {
139    return accepted;
140  }
