005package com.microfocus.silkcentral.democlient.gui;
006
007import java.awt.BorderLayout;
008import java.awt.Component;
009import java.awt.Frame;
010import java.awt.event.ActionEvent;
011import java.awt.event.ActionListener;
012import java.util.Collection;
013
014import javax.swing.BorderFactory;
015import javax.swing.DefaultListCellRenderer;
016import javax.swing.JButton;
017import javax.swing.JComboBox;
018import javax.swing.JDialog;
019import javax.swing.JLabel;
020import javax.swing.JList;
021import javax.swing.JPanel;
022
023import com.microfocus.silkcentral.democlient.gui.utils.UITool;
024import com.microfocus.silkcentral.webservices.tests.stubs.Project;
025
026/**
027 * Simple dialog for choosing a project.
028 */
029public class ProjectChooserDialog extends JDialog {
030
031  private static final long serialVersionUID = 1L;
032  private Project selectedProject;
033
034  public ProjectChooserDialog(Frame owner, Collection<Project> projects) {
035    super(owner, "Choose Project", true);
036    setDefaultCloseOperation(JDialog.DISPOSE_ON_CLOSE);
037    JPanel contentPane = (JPanel) getContentPane();
038    contentPane.setLayout(new BorderLayout(0, 10));
039    contentPane.setBorder(BorderFactory.createEmptyBorder(11, 11, 10, 10));
040
041    JButton okBtn = new JButton("OK");
042    JButton cancelBtn = new JButton("Cancel");
043
044    final JComboBox combo = new JComboBox();
045    for (Project project : projects) {
046      combo.addItem(project);
047    }
048
049    JPanel comboPn = new JPanel(new BorderLayout());
050    comboPn.add(new JLabel("Project:"), BorderLayout.WEST);
051    comboPn.add(combo, BorderLayout.CENTER);
052
053    combo.setRenderer(new DefaultListCellRenderer() {
054      @Override
055      public Component getListCellRendererComponent(JList list, Object value, int index, boolean isSelected, boolean cellHasFocus) {
056        JLabel lb = (JLabel) super.getListCellRendererComponent(list, value, index, isSelected, cellHasFocus);
057        Project project = (Project) value;
058        lb.setText(project.getName());
059        return lb;
060      }
061
062    });
063
064    contentPane.add(UITool.createButtonsPane(okBtn, cancelBtn), BorderLayout.SOUTH);
065    contentPane.add(comboPn, BorderLayout.NORTH);
066
067    cancelBtn.addActionListener(new ActionListener() {
068      @Override
069      public void actionPerformed(ActionEvent e) {
070        setVisible(false);
071        dispose();
072      }
073    });
074
075    okBtn.addActionListener(new ActionListener() {
076      @Override
077      public void actionPerformed(ActionEvent event) {
078        selectedProject = (Project) combo.getSelectedItem();
079        setVisible(false);
080        dispose();
081      }
082    });
083
084    pack();
085  }
086
087  public Project getSelectedProject() {
088    return selectedProject;
089  }
090
091}
